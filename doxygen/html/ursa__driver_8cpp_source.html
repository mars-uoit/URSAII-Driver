165  void Interface::processData() {
166  while (rx_buffer_.size() >= 3)
167  {
168  if (rx_buffer_.front() == 0xff)
169  {
170  rx_buffer_.pop_front(); //drop the first byte it is only for sync

174  char1 = rx_buffer_.front();
175  rx_buffer_.pop_front();
176  char2 = rx_buffer_.front();
177  rx_buffer_.pop_front();
178

181 
182  if (count == 0)
183  processBatt(energy);

187 #ifdef DEBUG_

191 #endif
192  pulses_[energy] += (count >> 2);
193  }
194  }
